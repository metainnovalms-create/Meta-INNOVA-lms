```typescript
// Frontend sync pattern using localStorage (development)
// Will be replaced with API calls in production

// localStorage keys
const STORAGE_KEYS = {
  COURSES: 'meta_courses',
  LEVELS: 'meta_course_levels',
  SESSIONS: 'meta_course_sessions',
  CONTENT: 'meta_course_content',
  COMPLETIONS: 'meta_content_completions',
  ASSIGNMENTS: 'meta_course_assignments',
  ENROLLMENTS: 'meta_course_enrollments'
};

// Load functions (always read fresh data)
export const loadCourses = (): Course[] => {
  const data = localStorage.getItem(STORAGE_KEYS.COURSES);
  return data ? JSON.parse(data) : [];
};

export const loadContentCompletions = (): ContentCompletion[] => {
  const data = localStorage.getItem(STORAGE_KEYS.COMPLETIONS);
  return data ? JSON.parse(data) : [];
};

// Save functions (persist and broadcast)
export const saveCourses = (courses: Course[]): void => {
  localStorage.setItem(STORAGE_KEYS.COURSES, JSON.stringify(courses));
  // In production: POST /api/v1/courses
};

export const saveContentCompletion = (completion: ContentCompletion): void => {
  const completions = loadContentCompletions();
  
  // Check if already exists
  const existingIndex = completions.findIndex(
    c => c.student_id === completion.student_id && 
         c.content_id === completion.content_id
  );
  
  if (existingIndex >= 0) {
    completions[existingIndex] = completion;
  } else {
    completions.push(completion);
  }
  
  localStorage.setItem(STORAGE_KEYS.COMPLETIONS, JSON.stringify(completions));
  // In production: POST /api/v1/content/{id}/complete
};

// Batch completion for officer
export const markContentCompleteForClass = async (params: {
  contentId: string;
  officerId: string;
  officerName: string;
  classId: string;
  sessionDate: string;
  presentStudentIds: string[];
}): Promise<number> => {
  const completions: ContentCompletion[] = params.presentStudentIds.map(studentId => ({
    id: generateUUID(),
    content_id: params.contentId,
    student_id: studentId,
    course_id: getContentCourseId(params.contentId),
    completed: true,
    completed_at: new Date().toISOString(),
    completed_by: 'officer',
    officer_id: params.officerId,
    officer_name: params.officerName,
    session_date: params.sessionDate,
    class_id: params.classId
  }));
  
  // Save all completions
  const existing = loadContentCompletions();
  const merged = [...existing, ...completions];
  localStorage.setItem(STORAGE_KEYS.COMPLETIONS, JSON.stringify(merged));
  
  return completions.length;
  // In production: POST /api/v1/content/{id}/mark-complete-for-class
};
```

```typescript
/**
 * Delete all content for a course (use with caution)
 */
export async function deleteCourseContent(courseId: string): Promise<number> {
  let deletedCount = 0;
  let continuationToken: string | undefined;
  
  do {
    const listResponse = await r2Client.send(new ListObjectsV2Command({
      Bucket: BUCKET_NAME,
      Prefix: `courses/${courseId}/`,
      ContinuationToken: continuationToken,
    }));
    
    if (listResponse.Contents) {
      for (const object of listResponse.Contents) {
        if (object.Key) {
          await deleteFile(object.Key);
          deletedCount++;
        }
      }
    }
    
    continuationToken = listResponse.NextContinuationToken;
  } while (continuationToken);
  
  return deletedCount;
}

 * Validate file type and size
 */
export function validateUpload(params: {
  fileName: string;
  contentType: string;
  size: number;
  allowedTypes: string[];
  maxSizeMB: number;
}): { valid: boolean; error?: string } {
  const extension = path.extname(params.fileName).toLowerCase();
  
  if (!params.allowedTypes.includes(extension)) {
    return {
      valid: false,
      error: `File type ${extension} not allowed. Allowed: ${params.allowedTypes.join(', ')}`,
    };
  }
  
  const maxSizeBytes = params.maxSizeMB * 1024 * 1024;
  if (params.size > maxSizeBytes) {
    return {
      valid: false,
      error: `File size ${(params.size / 1024 / 1024).toFixed(2)}MB exceeds limit of ${params.maxSizeMB}MB`,
    };
  }
  
  return { valid: true };
}
```

```typescript
import { useState } from 'react';
import axios from 'axios';
import { toast } from 'sonner';

interface UploadProgress {
  loaded: number;
  total: number;
  percentage: number;
}

interface UploadResult {
  success: boolean;
  publicUrl?: string;
  fileKey?: string;
  error?: string;
}

export const useContentUpload = () => {
  const [isUploading, setIsUploading] = useState(false);
  const [progress, setProgress] = useState<UploadProgress | null>(null);
  
  const uploadContent = async (params: {
    file: File;
    courseId: string;
    levelId: string;
    sessionId: string;
    onProgress?: (progress: UploadProgress) => void;
  }): Promise<UploadResult> => {
    setIsUploading(true);
    setProgress({ loaded: 0, total: params.file.size, percentage: 0 });
    
    try {
      // Step 1: Get presigned URL
      const presignedResponse = await axios.post('/api/v1/content/upload-url', {
        courseId: params.courseId,
        levelId: params.levelId,
        sessionId: params.sessionId,
        fileName: params.file.name,
        contentType: params.file.type,
        fileSize: params.file.size,
      });
      
      if (!presignedResponse.data.success) {
        throw new Error(presignedResponse.data.error);
      }
      
      const { uploadUrl, fileKey, publicUrl } = presignedResponse.data.data;
      
      // Step 2: Upload directly to R2
      await axios.put(uploadUrl, params.file, {
        headers: {
          'Content-Type': params.file.type,
        },
        onUploadProgress: (progressEvent) => {
          if (progressEvent.total) {
            const percentage = Math.round((progressEvent.loaded * 100) / progressEvent.total);
            const uploadProgress = {
              loaded: progressEvent.loaded,
              total: progressEvent.total,
              percentage,
            };
            setProgress(uploadProgress);
            params.onProgress?.(uploadProgress);
          }
        },
      });
      
      setIsUploading(false);
      setProgress(null);
      
      return {
        success: true,
        publicUrl,
        fileKey,
      };
    } catch (error: any) {
      setIsUploading(false);
      setProgress(null);
      
      const errorMessage = error.response?.data?.error || error.message || 'Upload failed';
      toast.error(errorMessage);
      
      return {
        success: false,
        error: errorMessage,
      };
    }
  };
  
  return {
    uploadContent,
    isUploading,
    progress,
  };
};

    if (bytes < 1024) return `${bytes} B`;
    if (bytes < 1024 * 1024) return `${(bytes / 1024).toFixed(1)} KB`;
    return `${(bytes / (1024 * 1024)).toFixed(1)} MB`;
  };
  
  return (

        onDrop={handleDrop}
        onDragOver={(e) => e.preventDefault()}
      >
        <input
          ref={fileInputRef}
          type="file"
          accept={acceptedTypes}
          onChange={handleFileSelect}
          className="hidden"
        />
        
        <Upload className="mx-auto h-12 w-12 text-muted-foreground mb-4" />
        <p className="text-sm text-muted-foreground">
          Drag and drop file here or click to browse
        </p>
        <p className="text-xs text-muted-foreground mt-2">
          Supported: PDF, PPT, PPTX, MP4, MOV, WebM (Max {maxSizeMB}MB)
        </p>
      </div>
      
      {/* Selected File */}
      {selectedFile && (
        <div className="flex items-center justify-between bg-muted p-3 rounded-lg">
          <div className="flex items-center gap-3">
            <File className="h-5 w-5 text-primary" />
            <div>
              <p className="text-sm font-medium">{selectedFile.name}</p>
              <p className="text-xs text-muted-foreground">
                {formatFileSize(selectedFile.size)}
              </p>
            </div>
          </div>
          <Button
            variant="ghost"
            size="icon"
            onClick={() => setSelectedFile(null)}
            disabled={isUploading}
          >
            <X className="h-4 w-4" />
          </Button>
        </div>
      )}
      
      {/* Upload Progress */}
      {isUploading && progress && (
        <div className="space-y-2">

          <p className="text-xs text-muted-foreground text-center">
            Uploading... {progress.percentage}%
          </p>
        </div>
      )}
      
      {/* Upload Button */}
      {selectedFile && !isUploading && (
        <Button onClick={handleUpload} className="w-full">
          <Upload className="mr-2 h-4 w-4" />
          Upload Content
        </Button>
      )}
    </div>
  );
};
```

```typescript
// =====================================================

// File: src/utils/contentHelpers.ts

/**
 * Get content viewer URL/embed based on type
 */
export const getContentViewerUrl = (content: CourseContent): string => {
  switch (content.type) {
    case 'pdf':
      // Direct PDF URL for iframe embed
      return content.file_url!;
      
    case 'ppt':
    case 'pptx':
      // Office Online viewer for PPT files
      const pptUrl = encodeURIComponent(content.file_url!);
      return `https://view.officeapps.live.com/op/embed.aspx?src=${pptUrl}`;
      
    case 'video':
      // Direct video URL for HTML5 video player
      return content.file_url!;
      
    case 'youtube':
      // Convert YouTube URL to embed format
      const videoId = extractYouTubeId(content.youtube_url!);
      return `https://www.youtube-nocookie.com/embed/${videoId}?rel=0&modestbranding=1`;
      
    case 'simulation':
    case 'link':
      // External URL
      return content.external_link!;
      
    default:
      return '';
  }
};

/**
 * Extract YouTube video ID from various URL formats
 */
export const extractYouTubeId = (url: string): string | null => {
  const patterns = [
    /(?:youtube\.com\/watch\?v=)([a-zA-Z0-9_-]{11})/,
    /(?:youtube\.com\/embed\/)([a-zA-Z0-9_-]{11})/,
    /(?:youtu\.be\/)([a-zA-Z0-9_-]{11})/,
    /(?:youtube\.com\/v\/)([a-zA-Z0-9_-]{11})/,
  ];
  
  for (const pattern of patterns) {
    const match = url.match(pattern);
    if (match) return match[1];
  }
  
  return null;
};

/**
 * Get icon for content type
 */
export const getContentTypeIcon = (type: ContentType): LucideIcon => {
  const icons: Record<ContentType, LucideIcon> = {
    pdf: FileText,
    ppt: Presentation,
    video: Video,
    youtube: Youtube,
    simulation: Gamepad2,
    link: ExternalLink,
  };
  return icons[type] || File;
};

/**
 * Get color for content type
 */
export const getContentTypeColor = (type: ContentType): string => {
  const colors: Record<ContentType, string> = {
    pdf: 'text-red-500',
    ppt: 'text-orange-500',
    video: 'text-blue-500',
    youtube: 'text-red-600',
    simulation: 'text-purple-500',
    link: 'text-green-500',
  };
  return colors[type] || 'text-gray-500';
};
```

```sql
CREATE OR REPLACE FUNCTION refresh_course_analytics(p_course_id UUID)
RETURNS VOID AS $$
BEGIN
  INSERT INTO course_analytics (course_id, total_enrollments, average_progress, computed_at)
  SELECT 
    p_course_id,
    COUNT(*),
    AVG(progress_percentage),
    NOW()
  FROM course_enrollments

    total_enrollments = EXCLUDED.total_enrollments,
    average_progress = EXCLUDED.average_progress,
    computed_at = NOW();
END;
```

## 12. Security Considerations

### 12.1 Content Access Security

```typescript
// =====================================================
// Content Access Security Middleware
// =====================================================

/**
 * Verify student can access content based on:
 * 1. Student is enrolled in the course
 * 2. Student's class has access to the content's level
 * 3. Course is assigned to student's institution
 */
const verifyContentAccess = async (req, res, next) => {
  const { contentId } = req.params;
  const { user } = req;
  
  // Get content with level info
  const content = await prisma.courseContent.findUnique({
    where: { id: contentId },
    include: {
      level: true,
      course: true,
    },
  });
  
  if (!content) {
    return res.status(404).json({ error: 'Content not found' });
  }
  
  if (user.role === 'student') {
    // Check enrollment
    const enrollment = await prisma.courseEnrollment.findUnique({
      where: {
        course_id_student_id: {
          course_id: content.course_id,
          student_id: user.id,
        },
      },
    });
    
    if (!enrollment) {
      return res.status(403).json({ error: 'Not enrolled in this course' });
    }
    
    // Check level access
    if (content.level.order > enrollment.accessible_level_count) {
      return res.status(403).json({ 
        error: 'Content level not accessible',
        message: `This content is in Level ${content.level.order}, but you only have access to Levels 1-${enrollment.accessible_level_count}`,
      });
    }
  }
  
  if (user.role === 'officer') {
    // Verify officer is assigned to the institution that has this course
    const assignment = await prisma.courseAssignment.findFirst({
      where: {
        course_id: content.course_id,
        institution_id: user.institution_id,
      },
    });
    
    if (!assignment) {
      return res.status(403).json({ error: 'Course not assigned to your institution' });
    }
  }
  
  req.content = content;
  next();
};
```

```typescript
  courseId: string;
  levelId: string;
  sessionId: string;
  fileName: string;
  contentType: string;
  maxSize: number;
}) => {
  // Validate content type
  const allowedTypes = [
    'application/pdf',
    'application/vnd.ms-powerpoint',
    'application/vnd.openxmlformats-officedocument.presentationml.presentation',
    'video/mp4',
    'video/quicktime',
    'video/webm',
  ];
  
  if (!allowedTypes.includes(params.contentType)) {
    throw new Error('Content type not allowed');
  }
  
  const command = new PutObjectCommand({
    Bucket: BUCKET_NAME,
    Key: fileKey,
    ContentType: params.contentType,
    // Restrict to exact content type
    Conditions: [
      ['eq', '$Content-Type', params.contentType],
      ['content-length-range', 0, params.maxSize],
    ],
  });
  
  // Short expiration
  return getSignedUrl(r2Client, command, { expiresIn: 3600 });
};
```

```typescript
    where: { id: classId },

            orderBy: { order: 'asc' },
```
